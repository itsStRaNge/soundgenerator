- Phase Correlation Meter
    (https://github.com/michaelting/Phase_Correlation/blob/master/phase_corr.py)
    def phase_correlation(a, b):
        G_a = np.fft.fft2(a)
        G_b = np.fft.fft2(b)
        conj_b = np.ma.conjugate(G_b)
        R = G_a*conj_b
        R /= np.absolute(R)
        r = np.fft.ifft2(R).real
        return r
